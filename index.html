const API_URL = "https://script.google.com/macros/s/AKfycbxzSHIqzJJED92lcvLiiOeg1t08I7ypBvOeH4dRQqIWTk5BvswYFzPKztiQ4FoVGpDj/exec";


async function carregarDados() {

  const status = document.getElementById("statusAPI");


  try {

    const res = await fetch(API_URL);

    const data = await res.json();


    console.log("DADOS:", data);


    const select = document.getElementById("jogadores");

    select.innerHTML = "";


    if (!data.jogadoresPro) {

      status.innerText = "⚠️ jogadoresPro não encontrado";

      return;

    }


    data.jogadoresPro.forEach(j => {

      let opt = document.createElement("option");

      opt.value = j.ID_Jogador;

      opt.textContent = j.Nome;

      select.appendChild(opt);

    });


    status.innerText = "✅ Dados carregados com sucesso";


  } catch (erro) {

    console.error(erro);

    status.innerText = "❌ Erro ao conectar com API";

  }

}


carregarDados();
